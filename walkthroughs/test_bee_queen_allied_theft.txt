# Test bee_queen post-alliance honey access
# Scenario: After alliance, honey is freely available

# Navigate to bee_queen_clearing
go south
go south
go east
go west

# Complete 3-flower sequence to achieve allied state
@set actors.player.inventory = ["moonpetal"]
give moonpetal to queen
@set actors.player.inventory = ["frost_lily"]
give frost lily to queen
@set actors.player.inventory = ["water_bloom"]
give water bloom to queen
@expect "completes the exchange"

# Verify allied state
@assert actors.bee_queen.properties.trust_state.current == 3
@assert actors.bee_queen.properties.state_machine.current == "allied"

# Take honey - should succeed without hostility when allied
take honey
@expect "honey"

# Verify queen still allied (hostile reaction did NOT fire)
@assert actors.bee_queen.properties.state_machine.current == "allied"

# Summary: Allied state allows honey access without hostile reaction
