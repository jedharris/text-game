# Waystone Fragment Integration Test
# Tests: picking up fragments, using them on the waystone, tracking placement

# Setup: player starts at nexus_chamber (default start location)
# Waystone spirit is at nexus_chamber

# ============================================================
# Test 1: Look around to verify waystone is visible
# ============================================================
{"type":"command","action":{"verb":"look"}}
@expect "Waystone"

# ============================================================
# Test 2: Use ice shard on waystone (first fragment)
# ============================================================
@set actors.player.inventory = ["ice_shard"]

{"type":"command","action":{"verb":"inventory"}}
@expect "ice shard"

{"type":"command","action":{"verb":"use","object":"ice shard","indirect_object":"waystone"}}
@expect "fragment"
@assert extra.waystone_fragments contains "ice_shard"

# ============================================================
# Test 3: Place second fragment (alpha fang)
# ============================================================
@set actors.player.inventory = ["alpha_fang_fragment"]

{"type":"command","action":{"verb":"use","object":"alpha fang fragment","indirect_object":"waystone"}}
@expect "fragment"
@assert extra.waystone_fragments contains "alpha_fang_fragment"

# ============================================================
# Test 4: Place third fragment (water pearl)
# ============================================================
@set actors.player.inventory = ["water_pearl"]

{"type":"command","action":{"verb":"use","object":"water pearl","indirect_object":"waystone"}}
@expect "fragment"
@assert extra.waystone_fragments contains "water_pearl"

# ============================================================
# Test 5: Place fourth fragment (spore heart fragment)
# ============================================================
@set actors.player.inventory = ["spore_heart_fragment"]

{"type":"command","action":{"verb":"use","object":"spore heart fragment","indirect_object":"waystone"}}
@expect "fragment"
@assert extra.waystone_fragments contains "spore_heart_fragment"

# ============================================================
# Test 6: Place fifth and final fragment (town seal) - completes waystone
# ============================================================
@set actors.player.inventory = ["town_seal"]

{"type":"command","action":{"verb":"use","object":"town seal","indirect_object":"waystone"}}
@expect "restored"
@assert extra.waystone_complete == true
