# Swimming Skill Gate Test
# Tests: skill gate blocks entry, Jek teaches swimming, gate opens after learning

# Setup: player at flooded_plaza, no skills
@set actors.player.location = "flooded_plaza"

# Try to enter tidal_passage (requires swimming) — should be blocked
{"type":"command","action":{"verb":"go","object":"east"}}
{"type":"command","action":{"verb":"go","object":"east"}} # EXPECTED_FAILURE: skill gate blocks without swimming
@expect "drown"
@expect "swim"

# Go to survivor_camp to learn swimming from Jek
@set actors.player.location = "survivor_camp"

# Ask Jek about swimming
{"type":"command","action":{"verb":"ask","object":"jek","indirect_object":"swimming"}}
@expect "teaching"
@assert actors.player.properties.skills.basic_swimming == true

# Ask again — should say already learned
{"type":"command","action":{"verb":"ask","object":"jek","indirect_object":"swimming"}}
@expect "already know"

# Now try entering tidal_passage — should work
@set actors.player.location = "flooded_plaza"
{"type":"command","action":{"verb":"go","object":"east"}}
{"type":"command","action":{"verb":"go","object":"east"}}
@expect "Tidal Passage"
