# Curiosity Dealer Vex Trading and Information Broker Walkthrough
# Tests: Dialog handler, gift reactions, trading, information sales, trust gates

# Setup: Give player gold
@set actors.player.properties.gold = 1000

# Go to market square where Vex is located
@goto market_square

# Test 1: Initial dialog (low trust) - shows locked items
{"type":"command","action":{"verb":"ask","object":"vex","indirect_object":"trade"}}
@expect "trusted clients"

# Test 2: Ask about curiosities
{"type":"command","action":{"verb":"ask","object":"vex","indirect_object":"curious"}}
@expect "strange and unusual"

# Test 3: Try to get information (should be trust-gated)
{"type":"command","action":{"verb":"ask","object":"vex","indirect_object":"information"}}
@expect "better acquainted"

# Test 4: View shop menu with some trust
@set actors.curiosity_dealer_vex.properties.trust_state.current = 1

{"type":"command","action":{"verb":"ask","object":"vex","indirect_object":"trade"}}
@expect "pre-collapse tools"
@expect "150 gold"

# Test 5: Buy available item (ancient_tools at trust 1)
{"type":"command","action":{"verb":"ask","object":"vex","indirect_object":"buy ancient tools"}}
@expect "hands you"
@expect "150 gold"

# Verify gold was deducted (item may not show in inventory if no backing item entity)
@assert actors.player.properties.gold == 850

# Test 6: Information broker (requires trust 2)
@set actors.curiosity_dealer_vex.properties.trust_state.current = 2

# Verify information costs gold
@set actors.player.properties.gold = 1000
{"type":"command","action":{"verb":"ask","object":"vex","indirect_object":"information"}}
@expect "100 gold"
@assert actors.player.properties.gold == 900

# Test 7: Higher trust - command orb available in trade menu
{"type":"command","action":{"verb":"ask","object":"vex","indirect_object":"trade"}}
@expect "command orb"
@expect "200 gold"

# Test 8: Trust progression dialog (default response at trust >= 1)
{"type":"command","action":{"verb":"ask","object":"vex","indirect_object":"greetings"}}
@expect "Good to see you again"
