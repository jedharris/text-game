# === Phase 1: Get Bandages from Market Square ===

look
@expect "Nexus"

go south
@expect "Forest"

go east
@expect "Trail"

go south
@expect "Gate"

go south
@expect "Market"

take bandages
@expect "bandages"
@assert actors.player.inventory contains "bandages"

# === Phase 2: Return to Nexus ===

go north
@expect "Gate"

go north
@expect "Trail"

go west
@expect "Forest"

go north
@expect "Nexus"

# === Phase 3: Go to Survivor Camp, Get Quest + Swimming + Air Bladder ===

go east
@expect "Flooded"

go northwest
@expect "Survivor"

# Get quest from Mira
ask mira about help
@expect "Mira"

# Learn swimming from Jek
ask jek about swimming
@expect "swimming"
@assert actors.player.properties.skills.basic_swimming == true

# Pick up air bladder for Garrett rescue
take air bladder
@expect "bladder"
@assert actors.player.inventory contains "air_bladder"

# === Phase 4: Rescue Delvan (First Rescue) ===

go southeast
@expect "Flooded"

go south
@expect "Merchant"

go south
@expect "Warehouse"

look
@expect "Delvan"
@assert extra.delvan_commitment_created == true

# Use bandages to rescue Delvan
use bandages on delvan
@expect "bleeding"
@assert extra.delvan_rescued == true
@assert actors.merchant_delvan.properties.state_machine.current == "freed"

# === Phase 5: Rescue Garrett (Second Rescue) ===

look

# Use air bladder to rescue Garrett
use air bladder on garrett

# === Phase 6: Return to Camp to Check Quest Status ===
