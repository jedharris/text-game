# Delvan Rescue Success + Mira Quest Integration
# Tests: encounter → commitment → rescue → Mira quest completion → friendly state

# Grant swimming skill needed for submerged passages
@set actors.player.properties.skills = {"basic_swimming": true}

# First activate Mira's quest so rescue triggers quest completion
@set actors.player.location = "survivor_camp"
{"type":"command","action":{"verb":"ask","object":"mira","indirect_object":"help"}}
@expect "trapped"
@assert extra.mira_quest_active == true

# Navigate to Sunken District
@set actors.player.location = "flooded_plaza"
{"type":"command","action":{"verb":"go","object":"south"}}
{"type":"command","action":{"verb":"go","object":"south"}}

# Give player bandages for rescue
@set actors.player.inventory = ["bandages"]

# Encounter Delvan (triggers commitment)
{"type":"command","action":{"verb":"look"}}
@expect "Delvan"
@assert extra.delvan_commitment_created == true

# Rescue: bandages remove bleeding → on_rescue_success → Mira quest completes
{"type":"command","action":{"verb":"use","object":"bandages","indirect_object":"delvan"}}
@expect "stop Delvan's bleeding"
@expect "You came"

# Verify rescue
@assert extra.delvan_rescued == true

# Verify Mira quest completed (single rescue → friendly)
@assert extra.mira_quest_completed == true
@assert actors.camp_leader_mira.properties.state_machine.current == "friendly"
