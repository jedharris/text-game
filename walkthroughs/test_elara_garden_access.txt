# Healer Elara - Garden Access Control
# Tests: Garden access granted at trust 3+
# Scenario: healer_elara.md scenario 9

# Navigate to healer_elara location
{"type":"command","action":{"verb":"go","object":"south"}}
{"type":"command","action":{"verb":"go","object":"south"}}
{"type":"command","action":{"verb":"go","object":"east"}}
{"type":"command","action":{"verb":"go","object":"south"}}

# Verify we're at the right location
{"type":"command","action":{"verb":"look"}}
@expect "Elara"

# Try to access garden without trust (should be denied)
{"type":"command","action":{"verb":"go","object":"east"}}
@expect "private" or "not allowed" or "permission"

# Build trust to level 3
@set actors.healer_elara.trust_state.current = 3

# Ask about the garden
{"type":"command","action":{"verb":"ask","object":"elara","indirect_object":"garden"}}
@expect "herbs" or "plants" or "welcome"

# Now try to access garden with sufficient trust
{"type":"command","action":{"verb":"go","object":"east"}}
@expect "garden" or "herbs" or "plants"

# Verify we're in the garden
{"type":"command","action":{"verb":"look"}}
@expect "garden"

# Verify flag is set
@inspect extra.has_garden_access
@expect true

# Summary: Garden access control verified
# - Access denied below trust 3
# - Access granted at trust 3+
# - has_garden_access flag set
