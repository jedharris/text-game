# Healer Elara Service Walkthrough
# Tests: Healing service, dialog reactions, gossip reactions, trust gates

# Go to healer location
go north
go east

# Test 1: Talk to Elara
talk to elara
@expect "heal" OR "help"

# Test 2: Request healing (no conditions)
talk to elara about heal
@expect "seem well" OR "no need"

# Test 3: Add condition and try healing
@set player.properties.conditions = [{"type": "bleeding", "severity": 2}]

talk to elara about heal
@expect "tends to your wounds"
@expect "bleeding"

# Test 4: Verify condition removed
@assert "bleeding" not in [c.get("type") for c in player.properties.get("conditions", [])]

# Test 5: Try healing with low trust
@set healer_elara.properties.trust_state.current = -3

talk to elara about heal
@expect "don't trust"

# Test 6: Reset trust and try again
@set healer_elara.properties.trust_state.current = 0

talk to elara about medicine
@expect "healing" OR "help"
