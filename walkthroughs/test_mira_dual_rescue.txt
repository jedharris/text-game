# Mira Rescue Quest - Dual Rescue (Both Garrett AND Delvan)
# Tests: Both rescues → allied state + camp services unlocked
# Uses real item_use mechanics for both rescues

# Move player to survivor camp and accept quest
@set actors.player.location = "survivor_camp"

{"type":"command","action":{"verb":"ask","object":"mira","indirect_object":"help"}}
@expect "trapped"
@assert extra.mira_quest_active == true

# Rescue Delvan via real item use
@set actors.player.inventory = ["bandages"]
@set actors.player.location = "merchant_warehouse"

{"type":"command","action":{"verb":"use","object":"bandages","indirect_object":"delvan"}}
@expect "bleeding"
@assert extra.delvan_rescued == true

# Single rescue completes quest as "friendly"
@assert extra.mira_quest_completed == true
@assert actors.camp_leader_mira.properties.state_machine.current == "friendly"

# Rescue Garrett via real item use
@set actors.player.inventory = ["air_bladder"]
@set actors.player.location = "sea_caves"

{"type":"command","action":{"verb":"use","object":"air bladder","indirect_object":"garrett"}}
@expect "air bladder"
@assert extra.garrett_rescued == true

# Second rescue upgrades to allied + unlocks camp services
@assert actors.camp_leader_mira.properties.state_machine.current == "allied"
@assert extra.camp_services_unlocked == true
