# Herbalist Maren Trading System Walkthrough
# Tests: Dialog reactions, trading handler, trust-based pricing

# Setup: Give player gold for purchases
@set actors.player.properties.gold = 500

# Go to herbalist location (market_square)
go south

go south

# Test 1: Dialog with Maren shows shop menu
ask maren about trade
@expect "Maren shows you her wares"
@expect "healing"
@expect "gold"

# Test 2: Try to buy specific item
ask maren about silvermoss
@expect "50 gold"

# Test 3: Check insufficient gold (set low gold temporarily)
@set actors.player.properties.gold = 10
ask maren about silvermoss
@expect "don't have enough"

# Test 4: Successful purchase
@set actors.player.properties.gold = 500
ask maren about silvermoss
@expect "hands you"
@expect "silvermoss"

# Verify item added to inventory
inventory
@expect "silvermoss"

# Verify gold deducted
@assert actors.player.properties.gold < 500

# Test 5: Trust-based pricing
# Increase trust and check for discount
@set actors.herbalist_maren.properties.trust_state.current = 3

ask maren about trade
@expect "discount"

# Check that prices are lower with higher trust
ask maren about healing_herbs
# Should see reduced price due to trust discount

# Test 6: Multiple purchases
ask maren about warm_cloak
@expect "hands you"

# Verify both items in inventory
inventory
