# Meridian Nexus - Echo + Waystone Fragments - Full Arc
# Tests: Talk to Echo, get hints, place ice_shard and water_pearl fragments
#   on waystone, check progress
# Systems: dialog_reactions (Echo), item_use_reactions (waystone),
#   fragment tracking, spatial navigation
# No @set shortcuts (except for fragments that require quest completion)

# === Phase 1: Talk to Echo at Nexus ===

{"type":"command","action":{"verb":"look"}}
@expect "Nexus"

# Go up to observatory to reach keeper's quarters
{"type":"command","action":{"verb":"go","object":"up"}}
@expect "Observatory"

{"type":"command","action":{"verb":"go","object":"east"}}
@expect "Keeper"

# Talk to the Echo
{"type":"command","action":{"verb":"ask","object":"echo","indirect_object":"help"}}
@expect "hunter"

{"type":"command","action":{"verb":"ask","object":"echo","indirect_object":"waystone"}}
@expect "fragment"

{"type":"command","action":{"verb":"ask","object":"echo","indirect_object":"status"}}
@expect "souls"

# === Phase 2: Get Ice Shard from Frozen Reaches ===

@goto ice_caves

# Ice shard is frozen - needs 3 attempts to chip free
{"type":"command","action":{"verb":"take","object":"ice shard"}} # EXPECTED_FAILURE: first attempt
@expect "frozen"

{"type":"command","action":{"verb":"take","object":"ice shard"}} # EXPECTED_FAILURE: second attempt
@expect "chip"

{"type":"command","action":{"verb":"take","object":"ice shard"}}
@expect "free"
@assert actors.player.inventory contains "ice_shard"

# === Phase 3: Get Water Pearl from Sunken District ===

# Need swimming to reach deep_archive via merchant_warehouse
# Learn swimming first
@goto survivor_camp

{"type":"command","action":{"verb":"ask","object":"jek","indirect_object":"swimming"}}
@expect "swimming"
@assert actors.player.properties.skills.basic_swimming == true

@goto deep_archive

{"type":"command","action":{"verb":"take","object":"pearl"}}
@expect "pearl"
@assert actors.player.inventory contains "water_pearl"

# === Phase 4: Return to Nexus and Place Fragments ===

@goto nexus_chamber

# Place ice shard on waystone
{"type":"command","action":{"verb":"use","object":"ice shard","indirect_object":"waystone"}}
@expect "shard"

# Place water pearl on waystone
{"type":"command","action":{"verb":"use","object":"pearl","indirect_object":"waystone"}}
@expect "pearl"

# === Phase 5: Check Progress with Echo ===

{"type":"command","action":{"verb":"go","object":"up"}}
@expect "Observatory"

{"type":"command","action":{"verb":"go","object":"east"}}
@expect "Keeper"

{"type":"command","action":{"verb":"ask","object":"echo","indirect_object":"waystone"}}
@expect "fragment"
