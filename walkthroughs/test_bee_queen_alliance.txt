# Test bee_queen alliance through 3-flower trade
# Scenario: Complete trade sequence → allied state → theft allowed
# Success = 100% expected behaviors (including expected successes)

# Navigate to bee_queen_clearing
{"type":"command","action":{"verb":"go","object":"south"}}
{"type":"command","action":{"verb":"go","object":"south"}}
{"type":"command","action":{"verb":"go","object":"east"}}
{"type":"command","action":{"verb":"go","object":"west"}}
{"type":"command","action":{"verb":"look"}}

# Verify bee_queen present
@expect "Bee Queen"

# Give first flower
@set actors.player.inventory = ["moonpetal"]
{"type":"command","action":{"verb":"give","object":"moonpetal","indirect_object":"queen"}}
@expect "accepts"
@assert extra.bee_queen_flowers_traded contains "moonpetal"
@assert actors.bee_queen.properties.trust_state.current == 1

# Give second flower
@set actors.player.inventory = ["frost_lily"]
{"type":"command","action":{"verb":"give","object":"frost lily","indirect_object":"queen"}}
@expect "accepts"
@assert extra.bee_queen_flowers_traded contains "frost_lily"
@assert actors.bee_queen.properties.trust_state.current == 2

# Give third flower - should transition to allied
@set actors.player.inventory = ["water_bloom"]
{"type":"command","action":{"verb":"give","object":"water bloom","indirect_object":"queen"}}
@expect "completes the exchange"
@assert extra.bee_queen_flowers_traded contains "water_bloom"
@assert actors.bee_queen.properties.trust_state.current == 3

# Now verify honey is freely available (no hostile reaction)
{"type":"command","action":{"verb":"take","object":"honey"}}

# Success - allied state allows theft without hostility
