# Test Equipment System
# Tests use verb to equip items and remove verb to unequip them
# Uses natural language synonyms: wear, wield, hold, don, take off, put down
# Game: simple_game
# Equippable item: sword
# Non-equippable item: key

# Start location has sword on ground
{"type":"command","action":{"verb":"look"}}

# Test 1: Take the sword (should not auto-equip)
{"type":"command","action":{"verb":"take","object":"sword"}}
{"type":"command","action":{"verb":"inventory"}}

# Test 2: Use the sword to equip it
# Expected: "You equip the sword."
{"type":"command","action":{"verb":"use","object":"sword"}}

# Test 3: Try to use the sword again (already equipped)
# Expected: Falls through to on_use behavior (if any), or "You use the sword."
{"type":"command","action":{"verb":"use","object":"sword"}}

# Test 4: Remove the sword
# Expected: "You unequip the sword."
{"type":"command","action":{"verb":"remove","object":"sword"}}

# Test 5: Try to remove when not equipped
# Expected: "The sword is not currently equipped."
{"type":"command","action":{"verb":"remove","object":"sword"}}

# Test 6: Re-equip the sword with "use"
# Note: "wield" is a synonym for "use" in the text parser but JSON protocol
# sends verbs directly to handlers, so we use the canonical verb here
{"type":"command","action":{"verb":"use","object":"sword"}}

# Unequip for next test
{"type":"command","action":{"verb":"remove","object":"sword"}}

# Test 7: Test natural language synonym "take off" to unequip
# First equip again
{"type":"command","action":{"verb":"use","object":"sword"}}
# Then test "take off"
# Expected: Should map to "remove" and unequip
{"type":"command","action":{"verb":"remove","object":"sword"}}

# Test 8: Test non-equippable item (key)
# Navigate to hallway to get key
{"type":"command","action":{"verb":"go","object":"north"}}
{"type":"command","action":{"verb":"take","object":"key"}}

# Test 9: Try to use (equip) non-equippable item
# Expected: Key has no use handler, so this should fail
{"type":"command","action":{"verb":"use","object":"key"}}  # EXPECTED_FAILURE: key has no item_use_reaction and is not equippable

# Test 10: Try to remove non-equippable item
# Expected: "The key is not something you can unequip."
{"type":"command","action":{"verb":"remove","object":"key"}}  # EXPECTED_FAILURE: key is not equippable

# Test 11: Verify sword can be re-equipped
{"type":"command","action":{"verb":"go","object":"south"}}
{"type":"command","action":{"verb":"use","object":"sword"}}

# Final inventory check
{"type":"command","action":{"verb":"inventory"}}
