# Tests: gift_reactions and take_reactions for bee_queen

{"type":"command","action":{"verb":"go","object":"south"}}
{"type":"command","action":{"verb":"go","object":"east"}}
{"type":"command","action":{"verb":"go","object":"west"}}

# Verify location and bee_queen present
@expect "Bee Queen"
@expect "Clearing"

# TEST 1: gift_reactions - Invalid gift (non-flower)
# Give invalid item to test rejection
@set actors.player.inventory = ["venison"]
{"type":"command","action":{"verb":"give","object":"venison","indirect_object":"queen"}}

# Verify rejection message
@expect "not what she seeks"

# TEST 2: gift_reactions - Valid gift (flower)
# Give moonpetal flower
@set actors.player.inventory = ["moonpetal"]
{"type":"command","action":{"verb":"give","object":"moonpetal","indirect_object":"queen"}}

# Verify acceptance
@expect "accepts"
@expect "quivering with appreciation"
@assert actors.bee_queen.properties.trust_state.current == 1

# TEST 3: gift_reactions - Duplicate flower rejected
# Try to give moonpetal again
@set actors.player.inventory = ["moonpetal"]
{"type":"command","action":{"verb":"give","object":"moonpetal","indirect_object":"queen"}}

# Verify duplicate rejection
@expect "already received"

# TEST 4: take_reactions - Theft attempt (not allied)
# royal_honey is already in clearing
{"type":"command","action":{"verb":"take","object":"honey"}}

# Verify hostile reaction
@expect "furious buzzing"
@expect "hatred"
@assert extra.bee_trade_destroyed == true

# ✓ gift_reactions: Accepts valid flowers, rejects invalid items

# ✓ gift_reactions: Updates trust and tracks flowers traded
