# Merchant Delvan - Contacts Knowledge Fragment
# Tests: Delvan shares contacts at trust 4+

# Setup: Delvan rescued, mobile, at warehouse
@set actors.player.location = "merchant_warehouse"
@set extra.delvan_rescued = true
@set actors.merchant_delvan.properties.state_machine.current = "mobile"

# Low trust - refuses to share contacts
@set actors.merchant_delvan.properties.trust_state.current = 2
ask delvan about contacts
@expect "trust"

# High trust - reveals contacts
@set actors.merchant_delvan.properties.trust_state.current = 4
ask delvan about contacts
@expect "network"
@assert extra.knows_delvan_contacts == true

# Summary: Contacts knowledge fragment verified
# - Trust 4+ required for contacts
# - Lower trust gets refusal
# - delvan_contacts fragment granted at high trust
