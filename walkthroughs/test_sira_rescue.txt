# Hunter Sira Rescue Walkthrough
# Tests: encounter_reactions, item_use_reactions, condition_reactions, dialog_reactions

# Navigate to Hunter's Camp (Beast Wilds)
go south
go east
go east

# Test encounter_reactions: First look triggers encounter with Sira
look
@expect "Sira"
@expect "clock is now ticking"

# Give player healing items
@set actors.player.inventory = ["bandages", "healing_herbs"]

# Test item_use_reactions + condition_reactions: Use bandages to stop bleeding
use bandages on sira
@expect "bleeding slows, then stops"
ASSERT extra.sira_bleeding_stopped == true
ASSERT actors.hunter_sira.properties.conditions.bleeding == null

# Test item_use_reactions + condition_reactions: Use herbs to heal leg (triggers fully healed)
use healing_herbs on sira
@expect "saved my life"
ASSERT extra.sira_leg_healed == true
ASSERT extra.sira_healed == true
ASSERT actors.hunter_sira.properties.conditions.broken_leg == null

# Test dialog_reactions: Ask Sira about help
ask sira about help
@expect "Elara"

# Test dialog_reactions: Ask about wolves
ask sira about wolves
@expect "pack"
