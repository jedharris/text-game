# Hunter Sira Rescue Walkthrough
# Tests: encounter_reactions, item_use_reactions, condition_reactions, dialog_reactions
# Covers scenario 1 from npc_hunter_sira.md: Encounter + Quick Rescue

# Navigate to Hunter's Camp (Beast Wilds)
go south
go east
go east

# Encounter triggers automatically on entering Hunter's Camp
@expect "Sira"
@expect "clock is now ticking"

# Verify commitment created
@assert extra.sira_commitment_created == true
@assert extra.sira_first_encounter_turn != null

# Give player healing items
@set actors.player.inventory = ["bandages", "healing_herbs"]

# Use bandages to stop bleeding
use bandages on sira
@expect "bleeding slows, then stops"
@assert extra.sira_bleeding_stopped == true

# Use herbs to heal leg (bleeding already stopped, so this completes rescue)
@set actors.player.inventory = ["healing_herbs"]
use herbs on sira
@expect "set and splint the broken leg"
@expect "saved my life"
@assert extra.sira_leg_healed == true
@assert extra.sira_healed == true

# No death gossip generated (sira_died_with_player key absent from extra)

# Dialog still works after rescue
ask sira about help
@expect "Elara"

ask sira about wolves
@expect "pack"
