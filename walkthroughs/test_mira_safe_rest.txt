# Camp Leader Mira - Safe Rest Service
# Tests: Mira provides safe rest service when in allied state
# Scenario: camp_leader_mira.md scenario 7

# Setup: Set Mira to allied state and reduce player health
@set actors.camp_leader_mira.properties.state_machine.current = "allied"
@set actors.player.properties.health = 50
@set actors.player.properties.max_health = 100

# Navigate to survivor_camp
go east
go northwest

# Verify we're at survivor camp
look
@expect "Mira"

# Ask Mira about resting
ask mira about rest
@expect "rest"

# Verify health restored to max (100)
@assert actors.player.properties.health == 100

# Ask about rest again - should acknowledge already rested
ask mira about rest
@expect "rested"

# Summary: Camp safe rest service verified
# - Allied state required
# - Health restored to max_health
# - Works via dialog with Mira
