# Camp Leader Mira - Safe Rest Service
# Tests: Mira grants camp rest to trusted allies

# Setup: Player at survivor_camp, quest completed (friendly state)
@set actors.player.location = "survivor_camp"
@set extra.mira_quest_completed = true
@set actors.camp_leader_mira.properties.state_machine.current = "friendly"

# Ask about resting at camp - friendly state grants rest
{"type":"command","action":{"verb":"ask","object":"mira","indirect_object":"rest"}}
@expect "welcome"

@set actors.camp_leader_mira.properties.state_machine.current = "neutral"

{"type":"command","action":{"verb":"ask","object":"mira","indirect_object":"rest"}}
@expect "Prove yourself"

# Summary: Camp safe rest service verified
